#Command used to make PUZZLES table
create table PUZZLES(
puzzle_id INT NOT NULL AUTO_INCREMENT,
puzzle TEXT NOT NULL,
solution TEXT,
rating TEXT,
creation_time TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
primary key(puzzle_id)
);
